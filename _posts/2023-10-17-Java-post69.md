### 변수와 자료형

```js
/**
 * 변수와 자료형
 *
 * 기본타입
 * - undefined
 * - number
 * - string
 * - boolean
 *
 * 객체
 * - object
 * - function
 *
 */

console.log("=== 기본 타입() ===");
10; // 에러발생 X
// v; 에러발생
v = 10; // 에러발생 X
var u;
var u = undefined; // var u;와 동일, 자바스크립트는 정의와 선언을 구분하지 않는다.
var i = 10;
console.log(u + ":" + typeof u);
console.log(i + ":" + typeof i);
console.log(v + ":" + typeof v);

```
